package com.mathcards.db;

import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.ResultSet;
import java.sql.Statement;

/**
 * Small command-line utility to verify PostgreSQL connectivity and run a couple of sanity queries.
 *
 * Usage (env vars or system properties):
 *  - JDBC_URL (default: jdbc:postgresql://localhost:5432/mathcards)
 *  - DB_USER (default: mathuser)
 *  - DB_PASS (default: mathpass)
 */
public class DbCheck {
    public static void main(String[] args) {
        String jdbcUrl = firstNonEmpty(System.getProperty("JDBC_URL"), System.getenv("JDBC_URL"), "jdbc:postgresql://localhost:5432/mathcards");
        String user = firstNonEmpty(System.getProperty("DB_USER"), System.getenv("DB_USER"), "mathuser");
        String pass = firstNonEmpty(System.getProperty("DB_PASS"), System.getenv("DB_PASS"), "mathpass");

        System.out.println("Using JDBC URL: " + jdbcUrl);

        try (Connection conn = DriverManager.getConnection(jdbcUrl, user, pass);
             Statement st = conn.createStatement()) {

            System.out.println("Connected to database successfully.");

            // Simple count from users
            try (ResultSet rs = st.executeQuery("SELECT COUNT(*) FROM users")) {
                if (rs.next()) {
                    System.out.println("users count: " + rs.getInt(1));
                }
            } catch (Exception e) {
                System.out.println("Could not query users table (might be empty or missing). Message: " + e.getMessage());
            }

            // Show up to 5 equations
            try (ResultSet rs = st.executeQuery("SELECT id, user_id, term1, term2, operation, solution FROM equations LIMIT 5")) {
                System.out.println("equations (up to 5):");
                int rows = 0;
                while (rs.next()) {
                    rows++;
                    System.out.printf("  id=%d user_id=%s term1=%s term2=%s op=%s solution=%s\n",
                            rs.getInt("id"), rs.getObject("user_id"), rs.getObject("term1"), rs.getObject("term2"), rs.getObject("operation"), rs.getObject("solution"));
                }
                if (rows == 0) System.out.println("  (no rows)");
            } catch (Exception e) {
                System.out.println("Could not query equations table. Message: " + e.getMessage());
            }

            System.out.println("DB check finished.");

        } catch (Exception e) {
            System.err.println("Failed to connect or run checks: " + e.getMessage());
            e.printStackTrace(System.err);
            System.exit(2);
        }
    }

    private static String firstNonEmpty(String... vals) {
        for (String v : vals) {
            if (v != null && !v.isBlank()) return v;
        }
        return null;
    }
}
