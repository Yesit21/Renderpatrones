package com.mathcards.service;

import java.text.Normalizer;
import java.util.Locale;
import java.util.Random;

import org.springframework.stereotype.Service;

import com.mathcards.model.Equation;

/**
 * Servicio para generar ecuaciones matemáticas de diferentes niveles de dificultad.
 */
@Service
public class GenerateEquationService {

    private static final String[] VALID_OPERATIONS = {"+", "-", "*", "/"};
    private static final Random random = new Random();

    public Equation generateEquation(String level, String selectedOperation) {
        String lvl = normalizeLevel(level);
        String op  = normalizeOp(selectedOperation);

        return switch (lvl) {
            case "basico"   -> generateBasico(op);
            case "avanzado" -> generateAvanzado(op);
            case "experto"  -> generateExperto(op);
            default         -> defaultEquation();
        };
    }

    // ----------------- Helpers -----------------
    private String normalizeLevel(String level) {
        if (level == null) return "";
        String noAccents = Normalizer.normalize(level, Normalizer.Form.NFD)
                .replaceAll("\\p{M}+", "");
        return noAccents.trim().toLowerCase(Locale.ROOT);
    }

    private String normalizeOp(String op) {
        if (op == null) return "+";
        op = op.trim();
        return switch (op) {
            case "+", "-", "*", "/" -> op;
            default -> "+";
        };
    }

    // ----------------- Niveles -----------------
    private Equation generateBasico(String op) {
        int x = randomInRange(1, 10);

        switch (op) {
            case "+" -> {
                int term1 = randomInRange(1, 5);
                int term2 = randomInRange(1, 15);
                double result = term1 * x + term2;
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
            case "-" -> {
                int term1 = randomInRange(1, 5);
                int term2 = randomInRange(1, 10);
                double result = term1 * x - term2; // resta corregida
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
            case "*" -> {
                int term1 = randomInRange(1, 3);
                int term2 = randomInRange(1, 5);
                double result = term1 * x * term2;
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
            case "/" -> {
                int term2 = randomInRange(1, 3);
                int term1 = term2 * randomInRange(1, 5); // múltiplo para que sea exacta
                double result = (double) (term1 * x) / term2;
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
            default -> {
                int term1 = 1;
                int term2 = 1;
                double result = x + 1;
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
        }
    }

    private Equation generateAvanzado(String op) {
        int x     = randomInRange(5, 20);
        int term1 = randomInRange(5, 15);
        int term2 = randomInRange(10, 30);
        int term3 = randomInRange(5, 20);
        int term4 = randomInRange(5, 15);

        String op1 = VALID_OPERATIONS[random.nextInt(VALID_OPERATIONS.length)];
        String op2 = VALID_OPERATIONS[random.nextInt(VALID_OPERATIONS.length)];

        double intermediateResult = term1 * x;

        // aplicar op1 con term2
        intermediateResult = applyOperation(intermediateResult, term2, op1);

        // aplicar op2 con term3
        intermediateResult = applyOperation(intermediateResult, term3, op2);

        // AHORA sí usamos selectedOperation (op) con term4
        double result = applyOperation(intermediateResult, term4, op);

        return new Equation(term1, term2, term3, term4, op1, op2, result, op, "x", x);
    }

    private Equation generateExperto(String op) {
        int x = randomInRange(10, 30);

        switch (op) {
            case "+" -> {
                int term1 = randomInRange(10, 40);
                int term2 = randomInRange(20, 70);
                double result = term1 * x + term2;
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
            case "-" -> {
                int term1 = randomInRange(10, 40);
                int term2 = randomInRange(20, 60);
                double result = term1 * x - term2; // resta corregida
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
            case "*" -> {
                int term1 = randomInRange(5, 20);
                int term2 = randomInRange(5, 20);
                double result = term1 * x * term2;
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
            case "/" -> {
                int term2 = randomInRange(5, 15);
                int term1 = term2 * randomInRange(5, 20); // múltiplo para división exacta
                double result = (double) (term1 * x) / term2;
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
            default -> {
                int term1 = 1;
                int term2 = 1;
                double result = x + 1;
                return new Equation(term1, term2, null, null, null, null, result, op, "x", x);
            }
        }
    }

    /**
     * Genera un entero aleatorio entre min y max (ambos inclusive) usando el Random compartido.
     */
    private int randomInRange(int min, int max) {
        return min + random.nextInt(max - min + 1);
    }

    /**
     * Aplica la operación 'op' entre a y b y devuelve el resultado como double.
     */
    private double applyOperation(double a, double b, String op) {
        return switch (op) {
            case "+" -> a + b;
            case "-" -> a - b;
            case "*" -> a * b;
            case "/" -> a / b;
            default -> a + b;
        };
    }

    private Equation defaultEquation() {
        return new Equation(1, 1, null, null, null, null, 2, "+", "x", 1);
    }
}
